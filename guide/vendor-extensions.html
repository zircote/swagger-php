Vendor extensions â€‹

The specification allows for custom properties as long as they start with "x-". Therefore, all swagger-php annotations have an x property which accepts an array (map) and will unfold into "x-" properties.

php
<?php

namespace Openapi\Snippets\VendorExtensions\CustomProperty;

use OpenApi\Attributes as OA;

#[OA\Info(
    title: 'Example',
    version: '1.0.0',
    x: [
        'some-name' => 'a-value',
        'another' => 2,
        'complex-type' => [
            'supported' => [
                ['version' => '1.0', 'level' => 'baseapi'],
                ['version' => '2.1', 'level' => 'fullapi'],
            ],
        ],
    ],
)]
class OpenApiSpec
{
}
php
<?php

namespace Openapi\Snippets\VendorExtensions\CustomProperty;

use OpenApi\Annotations as OA;

/**
 * @OA\Info(
 *     title="Example",
 *     version="1.0.0",
 *     x={
 *         "some-name": "a-value",
 *         "another": 2,
 *         "complex-type": {
 *             "supported": {
 *                 {"version": "1.0", "level": "baseapi"},
 *                 {"version": "2.1", "level": "fullapi"},
 *             }
 *         }
 *     }
 * )
 */
class OpenApiSpec
{
}

Results in:

yaml
openapi: 3.0.0
info:
  title: Example
  version: 1
  x-some-name: a-value
  x-another: 2
  x-complex-type:
    supported:
      - version: "1.0"
        level: baseapi
      - version: "2.1"
        level: fullapi
